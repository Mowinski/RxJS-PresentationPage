class Sheep {
  constructor(public isRam: boolean) { }
}

const flock = Array.from({ length: 15 }, () => getRandomSheep());

const flockObservable = Observable.from(flock);
const onlyRamObservable = Observable.from(flock)
  .filter((sheep: Sheep) => sheep.isRam);

flockObservable.subscribe(
  (sheep: Sheep) => this.flockResult.push(sheep),
);
onlyRamObservable.subscribe(
  (sheep: Sheep) => this.ramResult.push(sheep),
);
