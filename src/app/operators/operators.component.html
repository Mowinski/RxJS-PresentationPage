class Sheep {
  static _id = 0;
  public id: number;
  constructor(public color: string) {
    this.id = Sheep._id++;
  }
}

const flock = Array.from(
  { length: 15 },
  () => Math.random() < 0.8 ? new Sheep('white') : new Sheep('black')
);

const simpleObservable = Observable.from(flock);
const filteredObservable = Observable.from(flock).filter((sheep: Sheep) => sheep.color !== 'black');

simpleObservable.subscribe((sheep: Sheep) => this.simpleResult.push(sheep));
filteredObservable.subscribe((sheep: Sheep) => this.filterResult.push(sheep));
